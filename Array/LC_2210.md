## 2210.Count Hills and Valleys in an Array 
## https://leetcode.com/problems/count-hills-and-valleys-in-an-array/description/

```java
class Solution {
    public int countHillValley(int[] nums) {
        // Step 1: Remove consecutive duplicates
        List<Integer> filtered = new ArrayList<>();
        filtered.add(nums[0]);
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                filtered.add(nums[i]);
            }
        }

        int count = 0;

        // Step 2: Check for hills and valleys
        for (int i = 1; i < filtered.size() - 1; i++) {
            int prev = filtered.get(i - 1);
            int curr = filtered.get(i);
            int next = filtered.get(i + 1);

            // Hill: curr > both neighbors
            // Valley: curr < both neighbors
            if ((curr > prev && curr > next) || (curr < prev && curr < next)) {
                count++;
            }
        }

        return count;
    }
}
